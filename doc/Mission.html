def squads_txt
  squads_txt = ''
  squads.each do |squad|
    if not squad.blank?
      squads_txt += squad.name
      squads_txt += ','
    end
  end
  squads_txt.chop! if squads_txt.last == ','
  return squads_txt
end

def strategy
  if @loaded_strategy.blank?
    mission_strategy.eval_script
    @loaded_strategy = mission_strategy
  end
  return mission_strategy
