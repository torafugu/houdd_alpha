def self.execute
  Trial.find_all_by_occurred_at(nil).each do |trial|
    prepare(trial)
    turn_num = 1
    until (trial.guard_defeated? or trial.intruder_defeated?) and 2 < turn_num
    #5.times do |turn_num|
      squad_setup(trial)
      #break if trial.active_squads.blank?
      p "-- proceed_turn " + (turn_num).to_s + " --"
      proceed_turn(trial)
      turn_num += 1
      break if turn_num > 100
    end
  end
end
